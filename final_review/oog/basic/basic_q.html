Conceptual

For the following questions, determine the time complexity of each function in terms of big-Theta (θ) notation.

Q1

    def one(n):
        for a in range(n):
            for b in range(n/2):
                for c in range(n/4):
                    print(a + b + c)

Q2

    def two(n):
        for a in range(n):
            for b in range(100000000):
                for c in range(n):
                    print(a + b + c)

Q3

    def three(n):
        while n > 1:
            result = n * n
            print(result)
            n = n / 10
        return False

Q4

    def four(lst):
        if len(lst) < 12345:
            return lst[0]
        return four(lst[1:])

Q5

    def five(n):
        def helper(x):
            return x + n
        return helper(n/2)

Q6

    def reverse(lst):
        if not lst:
            return []
        result = reverse(lst[1:])
        result.append(lst[0])
        return result
