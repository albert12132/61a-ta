Control statements

1. fibonacci()

def fibonacci(n):
    cur, next, i = 0, 1, 0
    while i < n:
        cur, next = next, cur + next
        i += 1
    return cur

2. digits_five()

def digits_five(n):
    total = 0
    while n > 0:
        total += n % 10
        n //= 10
    return total % 5 == 0

3. sum_fives()

def sum_fives(n):
    total, i = 0, 1
    while i <= n:
        if digits_five(i):
            total += i
    return total

Higher-order Functions

1. make_mod_n()

def make_mod_n(n):
    def mod_n(m):
        return m % n
    return mod_n
