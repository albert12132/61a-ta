Basics

1) Implement equal.

    Map solution

    def equal(t1, t2):
        if t1.is_leaf and t2.is_leaf:
            return t1.entry == t2.entry
        elif (not t1.left != not t2.left) or \
                (not t1.right != not t2.right):
            return False
        else:
            return equal(t1.left, t2.left) and equal(t1.right, t2.right)

2) Implement size

    def size(t):
        if t.is_leaf:
            return 1
        else:
            return 1 + size(t.left) + size(t.right)

3) Implement height

    def height(t):
        if not t or t.is_leaf:
            return 0
        else:
            return 1 + max(height(t.left), height(t.right))

4) Implement max_tree

    def max_tree(b):
        if b.is_leaf:
            return b.entry
        else:
            return max_tree(b.right)
