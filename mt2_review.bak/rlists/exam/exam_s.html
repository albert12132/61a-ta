Code writing

1) Implement alternate.

    def alternate(rlist):
        new = tmp = Rlist(rlist.first)
        while rlist.rest != empty and rlist.rest.rest != empty:
            tmp.rest = Rlist(rlist.rest.rest.first)
            tmp, rlist = tmp.rest, rlist.rest.rest
        return new

2) Implement reverse.

    def reverse(rlist):
        new = Rlist(rlist.first)
        while rlist.rest != empty:
            rlist = rlist.rest
            new = Rlist(rlist.first, new)
        return new

3) Implement deep_map.

    def deep_map(rlist, f):
        while rlist != empty:
            if type(rlist.first) == Rlist:
                deep_map(rlist.first, f)
            else:
                rlist.first = f(rlist.first)
            rlist = rlist.rest

4) Implement deep_max.

    def deep_max(rlist):
        n = None
        while rlist != empty:
            elem = rlist.first
            if type(rlist.first) == Rlist:
                elem = deep_max(rlist.first)
            if n == None or n < elem:
                n = elem
            rlist = rlist.rest
        return n
