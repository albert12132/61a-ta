What would Python print?

    >>> L = [1, 2, 3, 4, 5]
    >>> map(lambda x: x*2, L)
    ______
    >>> tuple(map(lambda x: x*2, L))
    ______
    >>> L
    ______
    >>> tuple(map(lambda x: 3, L))
    ______
    >>> tuple(map(lambda x, y: (y, x), [(1, 2), (3, 4)]))
    ______

    >>> filter(lambda x: x % 2 == 0, L)
    ______
    >>> tuple(filter(lambda x: x % 2 == 0, L))
    ______
    >>> L
    ______
    >>> tuple(filter(lambda x: False, L))
    ______

    >>> from functools import reduce
    >>> reduce(lambda x, y: x + y, L)
    ______
    >>> reduce(lambda x: x**2, L)
    ______
