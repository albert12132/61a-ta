1) Implement make_sassy_function

    def make_sassy_function(f, msg):
        sassy = True
        def sassy_f(*args):
            sassy = not sassy
            if sassy:
                return msg
            return f(*args)
        return sassy_f

2) Implement sentence_buffer

    def sentence_buffer():
        sentence = ''
        def buffer(word):
            nonlocal sentence
            sentence += word + ' '
            if word[-1] == '.':
                result, sentence = sentence, ''
                return result.strip()
        return buffer
